function showPanel(id, el){
  document.querySelectorAll('.panel').forEach(p=>p.classList.remove('active'));
  document.querySelectorAll('.nav-item').forEach(n=>n.classList.remove('active'));
  var panel = document.getElementById(id);
  if(panel) panel.classList.add('active');
  if(el) el.classList.add('active');
  if(id==='p_stats') chargerStats();
  if(id==='p_users') chargerTousUsers();
  if(id==='p_reseau') chargerReseauAdmin();
  if(id==='p_email_cfg') chargerEmailCfg();
}

function confirmerArretAdmin(){
  if(!confirm('Arreter le serveur STEG PV ?\n\nTous les utilisateurs seront deconnectes.')) return;
  fetch('/shutdown',{method:'POST'}).catch(function(){});
  setTimeout(function(){
    document.body.innerHTML = '<div style="display:flex;height:100vh;align-items:center;justify-content:center;flex-direction:column;font-family:Arial;background:#003366;color:#fff;">'
      + '<div style="font-size:50px;margin-bottom:16px;">&#10003;</div>'
      + '<h2>Serveur STEG PV arrete</h2>'
      + '<p style="margin-top:10px;opacity:.8;">Vous pouvez fermer cette fenetre.</p></div>';
  }, 1200);
}

function chargerReseauAdmin(){
  fetch('/api/network_info').then(function(r){return r.json();}).then(function(d){
    var div = document.getElementById('rn_ips');
    if(!div) return;
    if(!d.ips || !d.ips.length){
      div.innerHTML='<div style="color:#856404;background:#fff3cd;padding:10px;border-radius:6px;">Aucune IP reseau detectee.</div>';
      return;
    }
    var h = '';
    d.ips.forEach(function(ip){
      var url = 'http://' + ip + ':' + d.port;
      h += '<div style="background:#001f3f;color:#00ff88;font-family:monospace;font-size:16px;padding:12px 18px;border-radius:6px;margin-bottom:8px;cursor:pointer;" data-url="' + url + '" onclick="copyUrl(this,this.dataset.url)" title="Cliquer pour copier">'
         + url + '<span style="float:right;font-size:11px;opacity:.7;">Copier</span></div>';
    });
    div.innerHTML = h;
    var qrDiv = document.getElementById('rn_qr');
    if(qrDiv && d.urls && d.urls[0]){
      qrDiv.innerHTML = '<div style="font-size:12px;color:#555;margin-bottom:6px;">QR Code Android</div>';
      var wrap = document.createElement('div');
      wrap.style.cssText = 'background:#fff;padding:10px;display:inline-block;border-radius:8px;border:2px solid #003366;';
      qrDiv.appendChild(wrap);
      if(window.QRCode) new QRCode(wrap,{text:d.urls[0],width:160,height:160,colorDark:'#003366',colorLight:'#ffffff'});

  });
}

function chargerRegions(autoSel){
  return fetch('/api/admin/regions').then(r=>r.json()).then(data=>{
    ALL_REGIONS=data;

    if(!data.length){ el.innerHTML='<div style="color:#aaa;text-align:center;padding:20px;">Aucune région</div>'; return; }

function selRegion(id, nom){
  // Stocker dans window.G ET dans un backup global pour fiabilité maximale
  G.rid = parseInt(id);
  window._RID = parseInt(id);  // backup
  G.did = 0;
  window._DID = 0;
  const _lbl=document.getElementById('lblRegion');
  if(_lbl){ _lbl.textContent=nom; _lbl.dataset.rid=id; }
  const _dbl=document.getElementById('lblDistrict');
  if(_dbl){ _dbl.dataset.did=''; _dbl.textContent='sélectionnez un district'; }
  const _btn=document.getElementById('btnAddDistrict');
  if(_btn) _btn.style.display='inline-block';
  const _bu=document.getElementById('btnAddUser');
  if(_bu) _bu.style.display='none';
  const _ag=document.getElementById('listeAgents');
  if(_ag) _ag.innerHTML='<div style="color:#aaa;font-size:12px;text-align:center;padding:20px;">← Choisissez un district</div>';
  document.querySelectorAll('.region-btn').forEach(btn => {
    btn.classList.remove('sel');
    if(btn.getAttribute('onclick') && btn.getAttribute('onclick').includes('selRegion('+id+',')) btn.classList.add('sel');
  });
  chargerDistricts();
}

function getRegionIdSafe(){
  // Lire depuis toutes les sources dans l'ordre de priorité
  if(G.rid) return G.rid;
  if(window._RID) return window._RID;
  const lbl = document.getElementById('lblRegion');
  if(lbl && lbl.dataset.rid) return parseInt(lbl.dataset.rid);
  // Dernière chance: trouver la région sélectionnée dans la liste
  const sel = document.querySelector('.region-btn.sel');
  if(sel) {
    const m = sel.getAttribute('onclick').match(/selRegion\((\d+)/);
    if(m) return parseInt(m[1]);
  }
  return 0;
}

function saveUser(){
  const uid=document.getElementById('edit_user_id').value;
  const auths=[...document.querySelectorAll('.auth-check:checked')].map(cb=>cb.value).join(',');
  const body={
    nom:document.getElementById('inp_u_nom').value.trim().toUpperCase(),
    prenom:document.getElementById('inp_u_prenom').value.trim().toUpperCase(),
    username:document.getElementById('inp_u_username').value.trim(),

  };

  const method=uid?'PUT':'POST';

    .then(r=>r.json()).then(res=>{
      if(res.error){ alr('alrUser',res.error,false); return; }

// ── Tous les utilisateurs ────────────────────────────────────────────
function chargerTousUsers(){
  // Charger tous les districts puis tous les users
  Promise.all(ALL_REGIONS.map(r=>fetch(`/api/admin/districts/${r.id}`).then(x=>x.json())))
    .then(distByRegion=>{
      const allDists=distByRegion.flat();
      return Promise.all(allDists.map(d=>fetch(`/api/admin/users/${d.id}`).then(x=>x.json()).then(us=>us.map(u=>({...u,district_nom:d.nom})))));
    }).then(byDist=>{
      ALL_USERS=byDist.flat();
      filtrerUsers();
    });
}

function filtrerUsers(){
  const q=(document.getElementById('searchUser')?.value||'').toLowerCase();
  const list=q?ALL_USERS.filter(u=>((u.nom||'')+(u.prenom||'')+(u.username||'')+(u.district_nom||'')).toLowerCase().includes(q)):ALL_USERS;
  document.getElementById('tbodyUsers').innerHTML=list.map(u=>`<tr>
    <td><b>${u.prenom} ${u.nom}</b><br><span style="color:#888;font-size:10px;">${u.username}${u.matricule?' | '+u.matricule:''}</span></td>
    <td style="font-family:monospace;font-size:11px;">${u.username}</td>
    <td><span class="badge-role badge-${u.role_code}">${u.role_code}</span></td>
    <td style="font-size:11px;">${u.district_nom||'—'}</td>
    <td>${u.actif?'<span style="color:#28a745;font-weight:bold;">Actif</span>':'<span style="color:#dc3545;">Inactif</span>'}</td>
    <td>
      <button class="btn btn-r" style="padding:2px 7px;font-size:10px;" onclick="deleteUser(${u.id},'${(u.prenom+' '+u.nom).replace(/'/g,"\\'")}');setTimeout(chargerTousUsers,500);">✕</button>
    </td>
  </tr>`).join('') || '<tr><td colspan="6" style="text-align:center;color:#aaa;padding:20px;">Aucun utilisateur</td></tr>';
}

// ── Stats ────────────────────────────────────────────────────────────
function chargerStats(){
  Promise.all([
    fetch('/api/admin/regions').then(r=>r.json()),
  ]).then(([regions])=>{
    Promise.all(regions.map(r=>fetch(`/api/admin/districts/${r.id}`).then(x=>x.json()))).then(dists=>{
      const allD=dists.flat();
      Promise.all(allD.map(d=>fetch(`/api/admin/users/${d.id}`).then(x=>x.json()))).then(users=>{
        const allU=users.flat();
        document.getElementById('statsGrid').innerHTML=[
          {lbl:'Régions',val:regions.length,c:'#003366'},
          {lbl:'Districts',val:allD.length,c:'#17a2b8'},
          {lbl:'Utilisateurs',val:allU.length,c:'#28a745'},
          {lbl:'Actifs',val:allU.filter(u=>u.actif).length,c:'#fd7e14'},
        ].map(s=>`<div class="stat-box" style="background:${s.c};"><div class="val">${s.val}</div><div class="lbl">${s.lbl}</div></div>`).join('');
      });
    });
  });
}
